
# Start all infra containers (Postgres, Redis, Kafka, Zookeeper, MongoDB)
docker compose -f docker-compose.infra.yml up -d

# Verify they're healthy before proceeding
docker compose -f docker-compose.infra.yml ps

1. service-registry     (Eureka — all others register here)
2. auth-service         (port 8081 — propertize-service depends on it)
3. propertize           (port 8082 — core service)
4. employee-service     (port 8083)
5. api-gateway          (port 8080 — must be last, routes to all above)


# Terminal 1
cd service-registry && ./mvnw spring-boot:run -Dspring-boot.run.profiles=local

# Terminal 2 (after registry is up)
cd auth-service && ./mvnw spring-boot:run -Dspring-boot.run.profiles=local

# Terminal 3 (after auth-service is up)
cd propertize && ./mvnw spring-boot:run -Dspring-boot.run.profiles=local

# Terminal 4
cd employee-service && ./mvnw spring-boot:run -Dspring-boot.run.profiles=local

# Terminal 5 (last — after all services registered in Eureka)
cd api-gateway && ./mvnw spring-boot:run -Dspring-boot.run.profiles=local

cd propertize-front-end && npm run dev
